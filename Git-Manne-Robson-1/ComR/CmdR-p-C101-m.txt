require(ggplot2) 
Dados <- data.frame(nomes = c("N0","N1","N2","N3","N4","N5","N6","N7","N8")
, x <- c(40,45,45,42,42,35,35,25,22)
, y <- c(50,68,70,66,68,66,69,85,75)
, tipo <- c(4,4,4,4,6,4,4,4,4)
, tamanho <- c(1,1,1,1,2,1,1,1,1)
)
pdf('p-C101-m.pdf')
ggplot(Dados, aes(x,y)) + geom_point(aes(shape = factor(tipo),size =tamanho) ) + scale_size_continuous(range = c(3,4))+ scale_shape(solid = FALSE)+  geom_text(aes(label=nomes) , hjust=-0.2,vjust=0.5)
dev.off()
